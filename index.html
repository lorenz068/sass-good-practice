Fonctionnalités

Les variables!

Déclaration


$variable-globale: #FFF;

.un-selecteur {
    $variable-privee: #E25D20;
    color: $variable-privee;
    background: $variable-globale;
}

Utilisation


.un-selecteur {
  color: #E25D20;
  background: #FFF;
}

Les mixins

Déclaration


// Paramètre $width optionnel
@mixin truncate($width: 100%) {
    display: block;
    width: $width: 100%;
    white-space: nowrap;
    overflow: hidden;
    text-overflow: ellipsis;
}

Utilisation


// avec la largeur par défaut de 100%
.un-selecteur {
    @include truncate();
}

// avec une largeur différente
.un-selecteur {
    @include truncate(50%);
}

Imbrication

Déclaration


// /!\ à utiliser avec prudence /!\
.parent {
    .enfant {
        cle: valeur;
    }
}

Compilation


.parent .enfant { 
    cle: valeur;
}

"Débrication"

Déclaration


// /!\ à utiliser avec prudence /!\
.enfant {
    .parent & {
        cle: valeur;
    }
}

Après Compilation


.parent .enfant { 
    cle: valeur;
}

Préfixation

Déclaration


.prefixe {
    &-selecteur {
        cle: valeur;
    }
}

Après Compilation


.prefixe-selecteur {
    cle: valeur;
}

Déclarer par spécificité (ITCSS)

7 couches

Bootstrap

  • Seulement les modules utilisés
  • Par spécificité
  • Avant app pour overrides

Settings (Variables)


// ----------------------------------------------------------- /
//                           SETTINGS
// Global variables, site-wide settings, config switches, etc.
// ----------------------------------------------------------- /

// EXTERNAL
// Variables to compile Bootstrap
@import "External/bootstrap/_variables";

// APP
@import "App/Settings/_colors";
@import "App/Settings/_length";
@import "App/Settings/_text";
@import "App/Settings/_branding";

TOOLS (Mixins)


// ----------------------------------------------------------- /
//                           TOOLS 
// Site-wide mixins and functions
// ----------------------------------------------------------- /

// EXTERNAL
// Mixins to compile Bootstrap
@import "External/bootstrap/_mixins";

// APP
@import "App/Tools/_triangle";
@import "App/Tools/_text";
@import "App/Tools/_news-common";

GENERIC (Reset)


// ----------------------------------------------------------- /
//                           GENERIC 
// Low-specificity, far-reaching rulesets (e.g. resets).
// ----------------------------------------------------------- /

// EXTERNAL
// Soft reset to have similary style over all browsers
@import "External/bootstrap/_normalize";
@import "External/bootstrap/_print.scss";

Base (Tag Html)


// --------------------------------------------------------------- /
//                           BASE 
// HTML tag elements (e.g. a {}, p{}, blockquote {}, address {}).
// --------------------------------------------------------------- /

// EXTERNAL
// Border-box + img class utilities
@import "External/bootstrap/_scaffolding";
// Defined default html element + typrographic + text class utilities
@import "External/bootstrap/_type";

// APP
@import "App/Base/_elements";

Object


// --------------------------------------------------------------------------------- /
//                           OBJECTS
// Objects, abstractions, and design patterns, NO branding (e.g. .center {}, grid {}).
// --------------------------------------------------------------------------------- /

// EXTERNAL
// Add the great grid 
@import "External/bootstrap/_grid";

// APP
// Example of the BEM methodology;
@import "App/Objects/_center";
@import "App/Objects/_distribute";
@import "App/Objects/_truncate.scss";

Components


// ------------------------------------------------------------------- /
//                           COMPONENTS 
// Discrete, complete chunks of UI (e.g. .header, .tiles .carousel {}).
// ------------------------------------------------------------------- /

// EXTERNAL

// APP
@import "App/Components/_button";
@import "App/Components/_header";
@import "App/Components/_footer";
@import "App/Components/_breadcrumb";

TRUMPS (Shame) :'(


// ----------------------------------------------------------------------------------------- /
//                           TRUMPS 
// High-specificity, very explicit selectors. Overrides and helper classes (e.g. .hidden {}).
// ----------------------------------------------------------------------------------------- /

// EXTERNAL
@import "External/bootstrap/_utilities";
@import "External/bootstrap/_responsive-utilities";
// APP
@import "App/Trumps/_shame";

Media query

Déclaration


.selecteur {
    width: 100%;

    @media(max-width: 960px) {
        width: 50%;
    }
}

Compilation


.selecteur {
    width: 100%;
}

@media (max-width: 960px) {
    .selecteur {
        width: 50%;
    }
}

Pseudo class

Déclaration


.selecteur {
    width: 100%;

    &:hover {
        cursor: pointer;
    }
}

Compilation


.selecteur {
    width: 100%;
}

.selecteur:hover {
    cursor: pointer;
}
