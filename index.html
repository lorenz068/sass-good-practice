Fonctionnalités

Les variables!

Déclaration


$primary-color: #E25D20;

Utilisation


.un-selecteur {
    color: $primary-color;
}

Les mixins

Déclaration


// Snipet qui permet d'ajouter "..." à la fin d'une ligne
@mixin truncate($width: 100%) {
    display: block;
    width: $width: 100%;
    white-space: nowrap;
    overflow: hidden;
    text-overflow: ellipsis;
}

Utilisation


// avec la largeur par défaut de 100%
.un-selecteur {
    @include truncate();
}

// avec une largeur différente
.un-selecteur {
    @include truncate(50%);
}
