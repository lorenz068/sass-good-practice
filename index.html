Fonctionnalités

Les variables!

Déclaration


$primary-color: #E25D20;
$white: #FFFFFF;

Utilisation


.un-selecteur {
    color: $primary-color;
	background: $white;
}

Les mixins

Déclaration


// Limiter la taille d'une ligne et ajouter "..." à la fin
// Paramètre $width optionnel
@mixin truncate($width: 100%) {
    display: block;
    width: $width: 100%;
    white-space: nowrap;
    overflow: hidden;
    text-overflow: ellipsis;
}

Utilisation


// avec la largeur par défaut de 100%
.un-selecteur {
    @include truncate();
}

// avec une largeur différente
.un-selecteur {
    @include truncate(50%);
}

Imbrication

Déclaration


// /!\ à utiliser avec prudence /!\
.parent {
	.enfant {
		cle: valeur;
	}
}

Après Compilation


.parent .enfant { 
	cle: valeur;
}

"Débrication"

Déclaration


// /!\ à utiliser avec prudence /!\
.enfant {
	.parent & {
		cle: valeur;
	}
}

Après Compilation


.parent .enfant { 
	cle: valeur;
}

Préfixation

Déclaration


.prefixe {
	&-selecteur {
		cle: valeur;
	}
}

Après Compilation


.prefixe-selecteur {
    cle: valeur;
}
